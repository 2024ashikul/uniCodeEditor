using System;

class Calculate{
    public static int CalculateSum( params int[] numbers){
        int sum = 0;
        foreach(int i in numbers){
            sum+=i;
        }
        return sum;
    }

    public static void Main(){
        int sum1 = CalculateSum(1, 2, 3);       // sum1 should be 6
        int sum2 = CalculateSum(10, 20, 30, 40); // sum2 should be 100
        int sum3 = CalculateSum();             // sum3 should be 0
        Console.WriteLine(sum1);
        Console.WriteLine(sum2);
        Console.WriteLine(sum3);
    }
}
